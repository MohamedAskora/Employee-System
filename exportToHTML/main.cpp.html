#include<iostream>
using namespace std;

int main()
{
    int const MAX = 10000;
    string name[MAX]={};
    double age[MAX]={0};
    int salary[MAX]={};
    char gender[MAX]={};
    int num=0;
    while (true)
    {
        cout << "Enter your choice:\n";
        cout << "1) Add new employee\n";
        cout << "2) Print all employees\n";
        cout << "3) Delete by age\n";
        cout << "4) Update salary by name\n";
        cout << "5) Exit\n";
        int choice;
        cin >> choice;
        if (choice<1 || choice>5)
        {
            cout << "Wrong Choice, plese enter a vaild choice\n";
            continue;
        }
        else if (choice==1)
        {
            cout << "Enter name: ";
            cin >> name[num];
            cout << "Enter age: ";
            cin >> age[num];
            cout << "Enter salary: ";
            cin >> salary[num];
            cout << "Enter gender (M/F): ";
            cin >> gender[num];
            num ++;
        }
        else if (choice==2)
        {
            cout << "************************\n";
            for (int j=0;j<num;j++)
            {
                if (age[j]==0)
                    continue;
                cout << name[j] <<" ";
                cout << age[j] <<" ";
                cout << salary[j] <<" ";
                cout << gender[j] <<"\n";
            }
        }
        else if (choice==3)
        {
            int start, end;
            cout << "Enter start and end age: ";
            cin >> start;
            cin >> end;
            for (int j=0;j<num;j++)
            {
                if (age[j]>=start && age[j]<=end)
                {
                    name[j]="";
                    age[j]=0;
                    salary[j]=0;
                }
            }
        }
        else if (choice==4)
        {
            string n;
            int s;
            cout << "Enter the name and salary ";
            cin >> n;
            cin >> s;
            bool is_found = false;
            for (int j=0;j<num;j++)
            {
                if (name[j]==n)
                {
                    is_found = true;
                    salary[j]=s;
                    break;
                }
            }
            if (!is_found)
                cout << "No employee with this name\n";
        }
        else if (choice==5)

            {
            return 0;
            }
    }


}
